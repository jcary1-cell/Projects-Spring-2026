    <script>

        // State Management

        let timeLeft = 3000; 

        let timerId = null;

        let currentGenre = "Academic Essay";

        const phaseNames = ["Synthesis", "De-Essay", "Drafting", "Signature"];


        // Timer Logic

        function updateTimer() {

            const mins = Math.floor(timeLeft / 60);

            const secs = timeLeft % 60;

            document.getElementById('timer-display').textContent = 

                `${mins.toString().padStart(2, '0')}:${secs.toString().padStart(2, '0')}`;

            

            if (timeLeft <= 0) {

                clearInterval(timerId);

                document.getElementById('timer-display').classList.add('text-red-500');

                return;

            }

            timeLeft--;

        }


        function toggleTimer() {

            const btn = document.getElementById('timer-btn');

            if (!timerId) {

                timerId = setInterval(updateTimer, 1000);

                btn.textContent = "Pause";

            } else {

                clearInterval(timerId);

                timerId = null;

                btn.textContent = "Resume";

            }

        }


        function resetTimer() {

            clearInterval(timerId);

            timerId = null;

            timeLeft = 3000;

            document.getElementById('timer-display').textContent = "50:00";

            document.getElementById('timer-btn').textContent = "Start";

        }


        // Navigation

        function showStep(num) {

            document.querySelectorAll('.step-card').forEach(s => s.classList.remove('active'));

            document.getElementById(`step-${num}`).classList.add('active');

            document.getElementById('phase-indicator').textContent = `Phase ${num}: ${phaseNames[num-1]}`;

            

            for (let i = 1; i <= 4; i++) {

                const btn = document.getElementById(`nav-${i}`);

                if (i === num) {

                    btn.className = "w-8 h-8 rounded-full bg-indigo-600 text-white text-sm font-bold ring-4 ring-white transition-all scale-110";

                } else if (i < num) {

                    btn.className = "w-8 h-8 rounded-full bg-emerald-500 text-white text-sm font-bold ring-4 ring-white transition-all";

                } else {

                    btn.className = "w-8 h-8 rounded-full bg-slate-200 text-slate-500 text-sm font-bold ring-4 ring-white transition-all";

                }

            }

            window.scrollTo(0,0);

        }
